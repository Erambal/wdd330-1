    Remember This:

    • for (const key of Object.keys (object_name) )
      -- gets you all of the attributes in the object
    • for( const [key , value] of Object.entries (superman) ) { console.log (` ${key} : ${value} `); }
      -- gets you a cleaner list
    • const dice = { sides : 6, roll() { return Math.floor ( this.sides * Math.random() ) + 1; } }
      -- creates an object 'dice' with six sides and a function to 'roll' and get random numbers
    • Math.abs()
      -- returns the absolute value
    • Math.ceil()
      -- returns the next highest integer
    • Math.floor()
      -- returns the next lowest integer
    • Math.round()
      -- returns the rounded integer
    • Math.pwo()
      -- returns the exponential, x^y
    • '?' makes the token optional
    • const pdf = /.*\.pdf$/; -- looks for a pdf
